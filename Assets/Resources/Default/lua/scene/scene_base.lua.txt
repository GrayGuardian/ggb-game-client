local SceneBase = class("SceneBase")
function SceneBase:init(info)
    print("Scene Init>>>info:" .. util.serialize(info))
    self.info = info
    timer:on(
        UE.Time.deltaTime,
        function()
            self:onLoad()
        end
    )
end
function SceneBase:onLoad()
    print("Scene onLoad")
end
function SceneBase:onDestroy()
    print("Scene onDestroy")
end

return SceneBase
