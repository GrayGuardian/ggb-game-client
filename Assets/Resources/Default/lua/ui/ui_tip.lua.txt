local base = require("ui_base")
local UITip = class("UITip", base)

base.static.abs = {  }

function UITip:onLoad()
    print("UI_Tip onLoad")
    self.msg = self.param.msg;
    self.uiComp.Text.msgText.text = self.msg
end
function UITip:onDestroy()
    print("UI_Tip onDestroy")
end
function UITip:onConfirmBtnClick(o,order)
    print('点击确定',self.msg,"  o:",o.name,"  order:",order);
end
return UITip
