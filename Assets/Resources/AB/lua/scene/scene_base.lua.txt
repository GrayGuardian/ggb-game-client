local SceneBase = class("SceneBase")
function SceneBase:init(info)
    print("Scene Init>>>info:" .. util.serialize(info))
    self.info = info
    timer:on(
        UE.Time.deltaTime,
        function()
            self:awake()
        end
    )
end
function SceneBase:awake()
    print("Scene awake")
end

return SceneBase
