require("base")
--主入口函数。从这里开始lua逻辑
function Main()
    print("logic start")
    SceneManager.LoadScene("Login")
end

--场景切换通知
function OnLevelWasLoaded(idx, name)
    collectgarbage("collect")
    Time.timeSinceLevelLoad = 0
    print("跳转场景>>>>", "idx:", idx, "name:", name)
    if idx >= 2 then
        require("scene"):new()
    end
end

function OnApplicationQuit()
    print("游戏退出")
end
