table.exist = function(table, value)
    for k, v in pairs(table) do
        if v == value then
            return true
        end
    end
    return false
end
table.del = function(table, value)
    for k, v in pairs(table) do
        if v == value then
            table[k] = nil
            return true
        end
    end
    return false
end
